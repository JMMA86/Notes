
```sql
CREATE TABLE TIPO_DE_NOTA (
    CODIGO NUMBER PRIMARY KEY,
    DESCRIPCION VARCHAR2(255)
);

CREATE TABLE ESTUDIANTE(
    CODIGO NUMBER PRIMARY KEY,
    NOMBRE VARCHAR(255) NOT NULL,
    USUARIO VARCHAR(255) NOT NULL,
    USUARIO_PADRE VARCHAR(255) NOT NULL
);

CREATE TABLE NOTAS (
    ID_NOTA NUMBER PRIMARY KEY,
    NOTA DECIMAL(2),
    CODIGO_TIPO NUMBER,
    CODIGO_ESTUDIANTE NUMBER,
    FOREIGN KEY (CODIGO_TIPO) REFERENCES TIPO_DE_NOTA(CODIGO),
    FOREIGN KEY (CODIGO_ESTUDIANTE) REFERENCES ESTUDIANTE(CODIGO)
);

```
